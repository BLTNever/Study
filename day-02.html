var arr1 = [1,23,4,5,6,7,3]; var arr2 = [2,34,,5,1,4,6,8]; function fn(){ var arr3 = arr1.concat(arr2); for(var i = 0; i < arr3.length; i++) { for(var j = i+1; j < arr3.length; j++) { if(arr3[i] === arr3[j])( arr3.splice(j,1); j--; ) } } return arr3; }
