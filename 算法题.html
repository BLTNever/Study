1. 斐波那契数列 

            迭代:
            function fibonacci1(n) {
                var num1 = 0, num2 = 1, num3, arr = [];
                for(var i = 3;i <= n;i++) {
                    num3 = num1 + num2;
                    num1 = num2;
                    num2 = num3;

                    arr.push(num3);
                }
                return arr;
            }
            fibonacci1(12);

            递归:
            function fibonacci2(n) {
                if(n <= 0) return 0;
                if(n <= 1) return 1;
                return fibonacci2(n - 1) + fibonacci2(n -2);
            }
            fibonacci2(12);
